302  bool setup() {
303  LOG_FDK(FDKDebug,__FUNCTION__);
304  AACENC_ERROR rc = AACENC_OK;
305 
306  // determine mode
307  switch (channels) {
308  case 1: mode = MODE_1; sce = 1; cpe = 0; break;
309  case 2: mode = MODE_2; sce = 0; cpe = 1; break;
310  case 3: mode = MODE_1_2; sce = 1; cpe = 1; break;
311  case 4: mode = MODE_1_2_1; sce = 2; cpe = 1; break;
312  case 5: mode = MODE_1_2_2; sce = 1; cpe = 2; break;
313  case 6: mode = MODE_1_2_2_1; sce = 2; cpe = 2; break;
314  default:
315  LOG_FDK(FDKError,"Unsupported WAV channels\n");
316  return false;
317  }
318 
319  // reopen encoder if modules or channels have changed
320  if (active && (channels>openChannels || openEncModules!=encModules)){
321  LOG_FDK(FDKWarning,"Basic Info has changed: we reopen the encoder\n");
322  rc = aacEncClose(&handle);
323  if (rc != AACENC_OK) {
324  LOG_FDK(FDKError,"Unable to close encoder: %s\n",setupErrorText(rc));
325  }
326  active = false;
327  }
328 
329  // open only once !
330  if (!active) {
331  LOG_FDK(FDKInfo,"aacEncOpen\n");
332  rc = aacEncOpen(&handle, encModules, channels);
333 
334  if (rc != AACENC_OK) {

336  return false;
337  }
338  // record open channels and modules
339  openChannels = channels;
340  openEncModules = encModules;
341

377  int updateParams() {
378  LOG_FDK(FDKDebug,__FUNCTION__);
379 
380  if (setParameter(AACENC_AOT, aot) != AACENC_OK) {
381  LOG_FDK(FDKError,"Unable to set the AOT (Audio Object Type)\n");

387  return -1;

391  LOG_FDK(FDKError,"Unable to set the AACENC_SAMPLERATE\n");

396  return -1;

401  }
402 
403  if (vbr) {

405  LOG_FDK(FDKError,"Unable to set the VBR bitrate mode\n");
406  return -1;
407  }
408  } else {
409  // determine bitrate
410  if (bitrate<=0){
411  if (aot==29){
412  sce = 1;
413  cpe = 0;
414  }
415  bitrate = (96*sce + 128*cpe) * sample_rate / 44;
416  if (aot==5 || aot== 29 || aot==132 || eld_sbr){
417  bitrate /= 2;
418  }
419  LOG_FDK(FDKWarning,"Determined bitrate: %d for sample rate %d\n", bitrate, sample_rate);
420  }
421 
422  if (bitrate>0) {
423  if (setParameter(AACENC_BITRATE, bitrate) != AACENC_OK) {
424  LOG_FDK(FDKError,"Unable to set the bitrate\n");
425  return -1;
426  }
427  }
428  }

430  LOG_FDK(FDKError,"Unable to set the ADTS transmux\n");
