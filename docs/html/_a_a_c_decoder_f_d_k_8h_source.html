188  size_t result = 0;

190  AAC_DECODER_ERROR error;

194  error = aacDecoder_Fill(aacDecoderInfo, (UCHAR **)&start, (const UINT*)&in_size, &bytesValid);
195  while (error == AAC_DEC_OK) {
196  error = aacDecoder_DecodeFrame(aacDecoderInfo, output_buffer, output_buffer_size, decoder_flags);
197  // write pcm to output stream

199  provideResult(output_buffer, output_buffer_size);

201  LOG_FDK(FDKError,"Decoding error: %d",error);
202  }
203  // if not all bytes were used we process them now

208  }

210  }
